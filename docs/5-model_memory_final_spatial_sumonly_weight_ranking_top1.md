```python
def MemoryLoss(memory):
    (m, d) = memory.size()
    memory_t = torch.t(memory)
    similarity = torch.matmul(memory, memory_t) / 2 + 1 / 2
    identity_mask = torch.eye(m).cuda()
    sim = torch.abs(similarity - identity_mask)
    return torch.sum(sim) / (m * (m - 1))
```
